```bash
# Validador canônico de links relativos (ignora fenced/inline code)
python3 <<'PY'
from pathlib import Path
import re, sys
root = Path("docs")
link_pattern = re.compile(r'!?\[[^\]]*\]\(([^)]+)\)')
fenced_pattern = re.compile(r'```.*?```', flags=re.DOTALL)
inline_pattern = re.compile(r'`[^`\n]*`')
broken = []
for source in root.rglob("*.md"):
    text = source.read_text(encoding="utf-8")
    text = fenced_pattern.sub("", text)
    text = inline_pattern.sub("", text)
    for raw in link_pattern.findall(text):
        target = raw.strip().split()[0].strip("<>")
        if not target or target.startswith(("http://","https://","mailto:","#","data:")):
            continue
        target = target.split("#", 1)[0]
        if not target: continue
        resolved = (source.parent / target).resolve()
        if not resolved.exists():
            broken.append((str(source), raw, str(resolved)))
if broken:
    for b in broken: print("BROKEN", b)
    sys.exit(1)
print("All relative Markdown links under docs/** resolve.")
PY
```
